insert(key, value, array) {
    s = length(array)
    hashValue = hash(key, s)
    i = 0
    while(i < s and array[(hashValue + i) mod s] != null)
        i += 1
    if(i < s) array[(hashValue + i) mod s] = (key, value)
}

search(key, array) {
    s = length(array)
    i = 0
    while(i < s) {
        if(array[i] = key) return i
        i += 1
    }
    return null
}
